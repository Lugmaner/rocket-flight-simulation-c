\section{Coordinate System and State Vector}

\begin{theorybox}{2D Inertial Frame}
We use a flat-Earth Cartesian frame (valid for ranges $\ll R_E = 6371\;\text{km}$):
\begin{itemize}
  \item $x$: horizontal downrange distance (m), positive in launch direction
  \item $y$: altitude above ground (m), positive upward
  \item Origin: launch pad
\end{itemize}
The complete state of the rocket at time $t$:
\[
  \mathbf{s}(t) = \begin{pmatrix} x \\ y \\ v_x \\ v_y \\ m \end{pmatrix}
  \in \mathbb{R}^5
\]
\end{theorybox}

\section{Forces Acting on the Rocket}

\subsection*{Gravity (altitude-dependent)}

\begin{theorybox}{Newton's Law of Gravitation}
\[
  g(y) = g_0\left(\frac{R_E}{R_E + y}\right)^2, \qquad g_0 = 9.80665\;\text{m/s}^2
\]
This follows from $F_g = G M_E m / r^2$ where $r = R_E + y$.\\
At $y = 0$: $g = 9.807\;\text{m/s}^2$. \;
At $y = 400\;\text{km}$ (ISS): $g \approx 8.69\;\text{m/s}^2$ — still 89\% of surface gravity!
\[
  \mathbf{F}_g = \begin{pmatrix} 0 \\ -mg(y) \end{pmatrix}
\]
\end{theorybox}

\subsection*{Thrust}

\begin{theorybox}{Rocket Thrust and the Momentum Equation}
Thrust arises from Newton's 3rd law: expelling mass backward creates forward force.
From momentum conservation (mass leaves at exhaust velocity $v_e$ relative to rocket):
\[
  T = \dot{m}_\text{fuel}\,v_e + (p_e - p_\text{atm})\,A_e
\]
For our model we use the simpler $T = \dot{m}\,v_e$ (vacuum thrust).
The thrust vector in 2D, controlled by pitch angle $\theta$ (measured from horizontal):
\[
  \mathbf{F}_T = T\begin{pmatrix}\cos\theta \\ \sin\theta\end{pmatrix}
\]
$\theta = 90°$: vertical thrust. $\theta = 0°$: horizontal thrust.
\end{theorybox}

\subsection*{Aerodynamic Drag}

\begin{theorybox}{Drag Vector}
Drag opposes the velocity vector $\mathbf{v} = (v_x, v_y)^\top$:
\[
  |\mathbf{v}| = \sqrt{v_x^2 + v_y^2}, \qquad \hat{\mathbf{v}} = \frac{\mathbf{v}}{|\mathbf{v}|}
\]
\[
  \mathbf{F}_D = -\underbrace{\frac{1}{2}\rho(y)\,|\mathbf{v}|^2\,C_d(Ma)\,A}_{F_D\ \text{(scalar)}}\cdot\hat{\mathbf{v}}
\]
\[
  F_{D,x} = -F_D\frac{v_x}{|\mathbf{v}|}, \qquad F_{D,y} = -F_D\frac{v_y}{|\mathbf{v}|}
\]
\textbf{Special case:} When $|\mathbf{v}| = 0$, drag is zero. Always check before dividing!
\end{theorybox}

\section{Equations of Motion — The Full ODE System}

Applying Newton's second law $\mathbf{F}_\text{total} = m\,\dot{\mathbf{v}}$ and adding mass continuity:

\begin{theorybox}{The 5-dimensional ODE}
\[
\frac{d\mathbf{s}}{dt} = \frac{d}{dt}
\begin{pmatrix} x \\ y \\ v_x \\ v_y \\ m \end{pmatrix}
=
\begin{pmatrix}
  v_x \\[0.3em]
  v_y \\[0.3em]
  \dfrac{T\cos\theta - F_D\,\dfrac{v_x}{|\mathbf{v}|}}{m} \\[0.8em]
  \dfrac{T\sin\theta - F_D\,\dfrac{v_y}{|\mathbf{v}|}}{m} - g(y) \\[0.8em]
  -\dot{m}
\end{pmatrix}
\]
This is our ODE: $\dot{\mathbf{s}} = f(t, \mathbf{s})$.\\
RK4 from Chapter 02 will integrate this 5D system forward in time.
\end{theorybox}

\subsection*{Non-dimensionalisation (advanced)}

Physical intuition improves when you identify the characteristic scales:
\[
  \tau = t\sqrt{g_0/R_E} \approx t/806\;\text{s}, \qquad
  \xi = x/R_E, \qquad
  V = v/\sqrt{g_0 R_E} \approx v/7905\;\text{m/s}
\]
The dimensionless orbital velocity is $V=1$ at the surface. This shows that LEO requires $V \approx 0.987$.

\section{Specific Impulse and the Rocket Equation}

\begin{theorybox}{Specific Impulse $I_{sp}$}
Instead of $v_e$, engineers use \textbf{specific impulse}:
\[
  I_{sp} = \frac{v_e}{g_0} \quad [\text{seconds}]
\]
RP-1/LOX (Falcon 9): $I_{sp} \approx 282\;\text{s}$ (sea level), $311\;\text{s}$ (vacuum)\\
LH2/LOX (Space Shuttle SSME): $I_{sp} \approx 366\;\text{s}$ (sea level), $453\;\text{s}$ (vacuum)\\
Ion thruster: $I_{sp} \approx 3000\;\text{s}$ (but tiny thrust!)

The Tsiolkovsky equation in terms of $I_{sp}$:
\[
  \Delta v = I_{sp}\,g_0\,\ln\!\left(\frac{m_0}{m_f}\right)
\]
\end{theorybox}

\section{Exercises}

\begin{exercisebox}{1 — Gravity at Altitude: Taylor Expansion}
Expand $g(y)$ around $y=0$ using the Taylor series to first order:
\[
  g(y) = g_0\left(1 + \frac{y}{R_E}\right)^{-2}
  \approx g_0\left(1 - \frac{2y}{R_E} + \cdots\right)
\]

\textbf{a)} Show the expansion step: $(1+\varepsilon)^{-2} \approx 1 - 2\varepsilon + 3\varepsilon^2 - \cdots$ for small $\varepsilon$.
This is the \textbf{binomial series}: $(1+x)^n = \sum_{k=0}^\infty \binom{n}{k}x^k$.

\textbf{b)} At what altitude $y^*$ does the first-order approximation $g \approx g_0(1-2y/R_E)$ differ from the exact formula by more than 1\%?
\[
  \left|\frac{g_0(1-2y/R_E) - g(y)}{g(y)}\right| = 0.01 \;\Rightarrow\; y^* \approx \underline{\hspace{2cm}}\;\text{km}
\]

\textbf{c)} Compute $g(y)$ exactly at $y = 0$, $100\;\text{km}$, $400\;\text{km}$, $36\,000\;\text{km}$ (GEO).
\end{exercisebox}

\begin{exercisebox}{2 — Force Balance at Launch}
Rocket at $t=0$: $m = 549\,054\;\text{kg}$, $T = 7\,607\;\text{kN}$, $\theta = 90°$, $v=0$.

\textbf{a)} Compute the thrust-to-weight ratio (TWR):
\[
  \text{TWR} = \frac{T}{m\,g_0} = \frac{7\,607\,000}{549\,054 \times 9.807} = \underline{\hspace{2cm}}
\]
Why must TWR $> 1$ for the rocket to lift off?

\textbf{b)} Net upward acceleration at launch ($F_D \approx 0$ since $v=0$):
\[
  a_y = \frac{T\sin 90° - 0}{m} - g_0 = \underline{\hspace{4cm}}\;\text{m/s}^2
\]

\textbf{c)} After 10 seconds (assuming constant thrust, $\dot{m} = 2530\;\text{kg/s}$):
\begin{align*}
  m(10) &= m_0 - \dot{m}\cdot 10 = \underline{\hspace{3cm}}\;\text{kg}\\
  v_y(10) &\approx a_y\cdot 10 = \underline{\hspace{3cm}}\;\text{m/s} \quad \text{(rough estimate)}\\
  Ma(10) &= v_y / c(0) \approx \underline{\hspace{3cm}}
\end{align*}
\end{exercisebox}

\begin{exercisebox}{3 — Drag Decomposition at an Angle}
At $t = 50\;\text{s}$: $v_x = 120\;\text{m/s}$, $v_y = 700\;\text{m/s}$, $y = 12\,000\;\text{m}$, $m = 42\,000\;\text{kg}$.
$A = 10.73\;\text{m}^2$.

\textbf{a)} Compute the velocity magnitude and direction:
\[
  |\mathbf{v}| = \sqrt{120^2 + 700^2} = \underline{\hspace{3cm}}\;\text{m/s}
\]
\[
  \alpha = \arctan\!\left(\frac{v_x}{v_y}\right) = \underline{\hspace{3cm}}° \quad\text{(angle from vertical)}
\]

\textbf{b)} Compute $Ma$, $C_d$, and total drag force $F_D$:
\[
  c(12000) = \sqrt{1.4 \times 287 \times T(12000)} = \underline{\hspace{3cm}}\;\text{m/s}
\]
\[
  Ma = \underline{\hspace{2cm}}, \quad C_d = \underline{\hspace{2cm}}, \quad \rho(12000) \approx \underline{\hspace{2cm}}\;\text{kg/m}^3
\]
\[
  F_D = \frac{1}{2}\times\underline{\hspace{1cm}}\times\underline{\hspace{2cm}}^2\times\underline{\hspace{1cm}}\times 10.73 = \underline{\hspace{3cm}}\;\text{N}
\]

\textbf{c)} Decompose drag into components:
\[
  F_{D,x} = F_D\cdot\frac{v_x}{|\mathbf{v}|} = \underline{\hspace{3cm}}\;\text{N}, \qquad
  F_{D,y} = F_D\cdot\frac{v_y}{|\mathbf{v}|} = \underline{\hspace{3cm}}\;\text{N}
\]

\textbf{d)} Compute accelerations $\dot{v}_x$ and $\dot{v}_y$ with $T = 7\,607\;\text{kN}$, $\theta = 80°$:
\[
  \dot{v}_x = \frac{T\cos(80°) - F_{D,x}}{m} = \underline{\hspace{4cm}}\;\text{m/s}^2
\]
\[
  \dot{v}_y = \frac{T\sin(80°) - F_{D,y}}{m} - g(12000) = \underline{\hspace{4cm}}\;\text{m/s}^2
\]
\end{exercisebox}

\begin{exercisebox}{4 — One Full RK4 Step on the 5D System}
Initial state: $\mathbf{s}_0 = [0, 0, 0, 0, 100\,000]^\top$.
Parameters: $T = 1\,400\,000\;\text{N}$, $\theta = 90°$, $\dot{m} = 437.5\;\text{kg/s}$, $A=10\;\text{m}^2$, $h_\text{step}=1\;\text{s}$.
Use $\rho(0) = 1.225$, $C_d = 0.3$, $g = 9.807$ (constants for this step).

\textbf{a)} Compute $\mathbf{k}_1 = f(\mathbf{s}_0)$:
\[
  \mathbf{k}_1 = [\dot{x},\; \dot{y},\; \dot{v}_x,\; \dot{v}_y,\; \dot{m}]
  = [0,\; 0,\; \underline{\hspace{1.5cm}},\; \underline{\hspace{1.5cm}},\; \underline{\hspace{1.5cm}}]^\top
\]
\textit{(Note: $|\mathbf{v}|=0$ at $t=0$, so $F_D=0$. Use $a_y = T/m - g$.)}

\textbf{b)} Compute the midpoint state $\mathbf{s}^* = \mathbf{s}_0 + \frac{1}{2}\mathbf{k}_1$. What is $m^*$? What is $v_y^*$?

\textbf{c)} For $\mathbf{k}_2$: now $|\mathbf{v}^*| = v_y^* \neq 0$. Compute $F_D$ at the midpoint state.
What is $\rho(y^*)$? What is the new $a_y$?

\textbf{d)} Why does $\dot{v}_y$ change between $k_1$ and $k_2$? Name all three reasons.
\end{exercisebox}

\begin{exercisebox}{5 — Orbital Velocity and Energy (Challenge)}
The first cosmic velocity (circular orbit just above the surface) satisfies:
\[
  \frac{mv^2}{R_E} = \frac{G M_E m}{R_E^2} = m\,g_0 \;\Rightarrow\; v_\text{orb} = \sqrt{g_0\,R_E}
\]

\textbf{a)} Compute $v_\text{orb}$ for Earth. What is the equivalent $Ma$ at sea level?

\textbf{b)} The specific orbital energy (energy per kg) of a circular orbit at altitude $h$:
\[
  E = -\frac{g_0 R_E^2}{2(R_E + h)}
\]
Compute $E$ at $h=0$, $h=400\;\text{km}$ (ISS), $h=36\,000\;\text{km}$ (GEO). Which orbit has more energy?

\textbf{c)} The delta-v needed to go from rest on the surface to LEO ($h=400\;\text{km}$, circular) is $\approx 9400\;\text{m/s}$.
The Tsiolkovsky ideal $\Delta v$ is $\approx 9200\;\text{m/s}$. Where does the extra $200\;\text{m/s}$ come from?
\end{exercisebox}
