\section{What is an Ordinary Differential Equation?}

\begin{theorybox}{Definition}
An \textbf{ordinary differential equation (ODE)} relates an unknown function $y(t)$ to its own derivatives:
\[
  \frac{dy}{dt} = f(t,\, y), \qquad y(t_0) = y_0
\]
The pair of the ODE and the initial condition $y(t_0)=y_0$ is called an \textbf{initial value problem (IVP)}.
The goal: find $y(t)$ for all $t > t_0$.
\end{theorybox}

\subsection*{Analytical Solution via Separation of Variables}

Many first-order ODEs can be solved exactly. The technique of \textbf{separation of variables} rewrites the ODE so that all $y$-terms are on one side and all $t$-terms on the other, then integrates both sides.

\textbf{Example:} $\dfrac{dy}{dt} = -k\,y$

\begin{align*}
  \frac{dy}{y} &= -k\,dt                   && \text{(separate)} \\
  \int \frac{1}{y}\,dy &= \int -k\,dt      && \text{(integrate both sides)} \\
  \ln|y| &= -kt + C                         && \text{(antiderivatives)} \\
  y(t) &= y_0\,e^{-kt}                      && \text{(exponentiate, apply IC)}
\end{align*}

This is the \textbf{exact} solution. Any numerical method must approximate this.

\section{The Taylor Series ‚Äî Foundation of All Numerical Methods}

\begin{theorybox}{Taylor Expansion of $y(t+h)$}
If $y(t)$ is smooth, we can expand it around $t$:
\[
  y(t+h) = y(t) + h\,\underbrace{y'(t)}_{= f(t,y)} + \frac{h^2}{2!}\,y''(t) + \frac{h^3}{3!}\,y'''(t) + \cdots
\]
This is an \textbf{exact} formula (infinite series). Every numerical method is derived by truncating this series at some order.
\end{theorybox}

\subsection*{Second Derivative via the Chain Rule}

The second derivative $y''(t)$ can be expressed using $f$:
\[
  y''(t) = \frac{d}{dt}f(t,y(t)) = \underbrace{\frac{\partial f}{\partial t}}_{\text{explicit}} + \underbrace{\frac{\partial f}{\partial y}\cdot f(t,y)}_{\text{implicit (chain rule)}}
\]
This is why higher-order Taylor methods require computing partial derivatives of $f$ ‚Äî which is expensive. Runge-Kutta methods (Chapter 02) achieve the same accuracy without this.

\begin{theorybox}{The Euler Method from Taylor Truncation}
Truncate after the first derivative term:
\[
  y(t+h) \approx y(t) + h\,f(t,y(t))
\]
\[
  \boxed{y_{n+1} = y_n + h\cdot f(t_n,\,y_n)}
\]
The \textbf{local truncation error} is the dropped term: $\dfrac{h^2}{2}y''(\xi) = \mathcal{O}(h^2)$.\\
After $N = T/h$ steps, errors accumulate: \textbf{global error} $= \mathcal{O}(h)$.
\end{theorybox}

\section{Geometric Interpretation}

At each point $(t_n, y_n)$, the ODE tells us the \textbf{slope} of the solution curve: $f(t_n, y_n)$.
Euler's method moves a distance $h$ along this slope ‚Äî a straight-line approximation.
The error arises because the true curve is not actually straight.

\section{Stability Analysis}

\begin{theorybox}{The Test Equation}
The canonical stability test equation is $\dot{y} = \lambda y$ with $\lambda \in \mathbb{C}$.\\
Applying Euler: $y_{n+1} = y_n + h\lambda y_n = (1 + h\lambda)\,y_n$

After $n$ steps: $y_n = (1+h\lambda)^n y_0$.\\
For stability we need $|1 + h\lambda| \leq 1$.

\textbf{Real case ($\lambda < 0$):} Stability requires $-2 \leq h\lambda \leq 0$, i.e.\
\[
  h \leq \frac{2}{|\lambda|}
\]
This is the \textbf{stability condition} ‚Äî the step size has an upper bound!
\end{theorybox}

\begin{exercisebox}{6 ‚Äî The Logistic Equation (Challenge)}
The logistic ODE models population growth with a carrying capacity $K$:
\[
  \frac{dy}{dt} = r\,y\left(1 - \frac{y}{K}\right), \quad y(0) = y_0
\]

\textbf{a)} Use \textbf{partial fractions} to separate and integrate:
\[
  \int \frac{dy}{y(1 - y/K)} = \int r\,dt
\]
\textit{Hint:} $\dfrac{1}{y(1-y/K)} = \dfrac{1}{y} + \dfrac{1/K}{1-y/K}$

\textbf{b)} Show that the exact solution is:
\[
  y(t) = \frac{K}{1 + \left(\frac{K}{y_0}-1\right)e^{-rt}}
\]

\textbf{c)} With $r=1$, $K=100$, $y_0=10$: Apply Euler with $h=0.5$ for 3 steps.
Compare to exact solution at $t=1.5$.

\textbf{d)} What happens to $y(t)$ as $t\to\infty$? Find the equilibrium analytically by setting $\dot{y}=0$.
