\section{Motivation: Euler ist erster Ordnung}

\begin{theorybox}{Das Problem mit Euler}
Der Euler-Schritt entspricht dem ersten Glied der Taylor-Reihe von $y(t+h)$:
\[
  y(t+h) = y(t) + h\,\dot{y}(t) + \underbrace{\frac{h^2}{2}\ddot{y}(t) + \cdots}_{\text{Euler ignoriert das!}}
\]
Der \textbf{lokale Fehler} ist $\mathcal{O}(h^2)$ — der abgeschnittene Term.\\
Nach $N = T/h$ Schritten akkumuliert sich ein \textbf{globaler Fehler} von $\mathcal{O}(h)$.

\textbf{Ziel:} Wir wollen ein Verfahren, das mehr Taylor-Glieder nutzt, ohne höhere Ableitungen $\ddot{y}, \dddot{y}$ explizit berechnen zu müssen.
\end{theorybox}

% ─────────────────────────────────────────────
\section{Die Idee: Mehrere Steigungen mitteln}

\begin{theorybox}{Grundprinzip von Runge-Kutta}
Statt nur die Steigung am Anfang des Schritts zu nehmen, berechnen wir \textbf{mehrere Steigungen} an verschiedenen Punkten innerhalb des Intervalls $[t_n,\, t_n+h]$ und bilden einen gewichteten Mittelwert.
\end{theorybox}

\subsection*{Die vier Steigungen von RK4}

\begin{theorybox}{RK4-Formel}
\[
  k_1 = f(t_n,\; y_n)
\]
\[
  k_2 = f\!\left(t_n + \tfrac{h}{2},\; y_n + \tfrac{h}{2}k_1\right)
\]
\[
  k_3 = f\!\left(t_n + \tfrac{h}{2},\; y_n + \tfrac{h}{2}k_2\right)
\]
\[
  k_4 = f\!\left(t_n + h,\; y_n + h\,k_3\right)
\]
\[
  \boxed{y_{n+1} = y_n + \frac{h}{6}\left(k_1 + 2k_2 + 2k_3 + k_4\right)}
\]
\end{theorybox}

\subsection*{Intuition hinter den Gewichten}
\begin{itemize}
  \item $k_1$: Steigung am \textbf{Anfang} des Schritts
  \item $k_2$: Steigung in der \textbf{Mitte}, ausgehend von $k_1$
  \item $k_3$: Steigung in der \textbf{Mitte}, ausgehend von $k_2$ (verfeinert)
  \item $k_4$: Steigung am \textbf{Ende}, ausgehend von $k_3$
  \item Gewichtung: $1:2:2:1$ — Mittelpunkte sind wichtiger (Simpsonregel!)
\end{itemize}

\begin{theorybox}{Fehlerordnung von RK4}
\begin{itemize}
  \item \textbf{Lokaler Fehler:} $\mathcal{O}(h^5)$
  \item \textbf{Globaler Fehler:} $\mathcal{O}(h^4)$
\end{itemize}
RK4 ist ein Verfahren \textbf{vierter Ordnung}. Halbiert man $h$, fällt der Fehler auf $1/16$ seines Wertes!
\end{theorybox}

\begin{exercisebox}{1 — RK4 von Hand: ein Schritt}
Gegeben: $\dfrac{dy}{dt} = -y$, $y(0) = 1$, $h = 0{,}5$.\\[0.5em]
Berechne \textbf{einen RK4-Schritt} und bestimme $y_1$.

Fülle schrittweise aus:

\begin{align*}
k_1 &= f(0{,}0,\; 1{,}0000) = \underline{\hspace{3cm}}\\[0.5em]
k_2 &= f\!\left(0{,}25,\; 1{,}0 + 0{,}25 \cdot k_1\right) = \underline{\hspace{3cm}}\\[0.5em]
k_3 &= f\!\left(0{,}25,\; 1{,}0 + 0{,}25 \cdot k_2\right) = \underline{\hspace{3cm}}\\[0.5em]
k_4 &= f\!\left(0{,}5,\; 1{,}0 + 0{,}5 \cdot k_3\right) = \underline{\hspace{3cm}}\\[1em]
y_1 &= 1{,}0 + \frac{0{,}5}{6}(k_1 + 2k_2 + 2k_3 + k_4) = \underline{\hspace{3cm}}

\textbf{Fehler RK4:} \underline{\hspace{2cm}} \quad \textbf{Fehler Euler:} \underline{\hspace{2cm}}
\end{exercisebox}

\begin{array}{c|cccc}

    & 1/6 & 2/6 & 2/6 & 1/6
\end{array}
