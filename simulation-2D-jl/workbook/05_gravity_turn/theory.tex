\section{The Tsiolkovsky Rocket Equation — Full Derivation}

\begin{theorybox}{Starting Point: Newton's 2nd Law for a Variable-Mass System}
Consider the rocket at time $t$ with mass $m$ and velocity $v$ (1D, no drag, no gravity for now).
In time $dt$, mass $dm_e = -dm > 0$ is expelled at velocity $v - v_e$ (absolute frame), where $v_e > 0$ is the exhaust speed relative to the rocket.

\textbf{Conservation of momentum:}
\[
  m\,v = (m + dm)(v + dv) + (-dm)(v - v_e)
\]
Expand and drop $dm\,dv$ (second order small):
\[
  m\,v = m\,v + m\,dv + v\,dm + dv\,dm - v\,dm + v_e\,dm
\]
\[
  0 = m\,dv + v_e\,dm
\]
\end{theorybox}

\begin{theorybox}{Integration: The Rocket Equation}
\[
  m\,dv = -v_e\,dm \quad\Rightarrow\quad dv = -v_e\,\frac{dm}{m}
\]
Integrate from $t=0$ (mass $m_0$, velocity $v_0$) to burnout (mass $m_f$, velocity $v_f$):
\[
  \int_{v_0}^{v_f} dv = -v_e\int_{m_0}^{m_f}\frac{dm}{m}
\]
\[
  v_f - v_0 = -v_e\,[\ln m]_{m_0}^{m_f} = -v_e(\ln m_f - \ln m_0) = v_e\ln\frac{m_0}{m_f}
\]
\[
  \boxed{\Delta v = v_e\ln\frac{m_0}{m_f}}
\]
This is one of the most important equations in space exploration.
\end{theorybox}

\begin{theorybox}{Losses: The Real $\Delta v$ Budget}
In reality, not all propulsive $\Delta v$ goes into useful speed:
\[
  \Delta v_\text{ideal} = \Delta v_\text{orbital} + \underbrace{\Delta v_\text{gravity}}_{\text{gravity drag}} + \underbrace{\Delta v_\text{aero}}_{\text{aero drag}} + \underbrace{\Delta v_\text{steer}}_{\text{steering losses}}
\]
For LEO ($\approx 9400\;\text{m/s}$ needed): typical split is
$7800$ orbital + $1200$ gravity drag + $200$ aero drag + $200$ steering.
\end{theorybox}

\section{The Gravity Turn}

\begin{theorybox}{Why Not Just Fly Straight Up?}
A purely vertical ascent maximizes altitude gain per fuel, but achieves no horizontal velocity.
Orbital insertion requires $v_x \approx 7800\;\text{m/s}$ — all horizontal.
The challenge: pitch over \textit{efficiently} (minimize steering losses) while building horizontal speed.
\end{theorybox}

\begin{theorybox}{Gravity Turn: The Elegant Solution}
After an initial kick at low altitude, the rocket's pitch angle follows the velocity vector:
\[
  \theta(t) = \arctan\!\left(\frac{v_y}{v_x}\right)
\]
Gravity continuously curves the velocity vector toward horizontal.\\[0.4em]
\textbf{Key property:} The angle of attack (between thrust and velocity) is always zero.
This eliminates aerodynamic side loads and steering losses.\\[0.4em]
\textbf{Physically:} The rocket is in ``free fall'' horizontally while thrust builds total speed.
\end{theorybox}

\subsection*{Linear Pitch Program (Simplified)}

When the velocity-vector tracking starts too early (before building sufficient horizontal speed), the trajectory may not reach orbit. A \textbf{scheduled pitch program} is more controllable:
\[
  \theta(t) = \begin{cases}
    \pi/2 & t \leq t_\text{kick} \\
    \pi/2 + (\theta_f - \pi/2)\dfrac{t - t_\text{kick}}{t_f - t_\text{kick}} & t_\text{kick} < t \leq t_f \\
    \theta_f & t > t_f
  \end{cases}
\]

\section{Gravity Loss Integral}

\begin{theorybox}{Quantifying Gravity Drag}
The gravity drag (velocity lost fighting gravity during vertical component of flight):
\[
  \Delta v_\text{grav} = \int_0^{t_\text{MECO}} g(y(t))\sin\theta(t)\,dt
\]
For constant $g$ and linear pitch from $90°$ to $0°$ over time $t_f$:
\[
  \Delta v_\text{grav} = g\int_0^{t_f}\sin\!\left(\frac{\pi}{2} - \frac{\pi}{2t_f}t\right)dt
  = g\int_0^{t_f}\cos\!\left(\frac{\pi t}{2t_f}\right)dt
\]
Evaluate using $\int\cos(ax)\,dx = \frac{1}{a}\sin(ax)$:
\[
  \Delta v_\text{grav} = g\cdot\frac{2t_f}{\pi}\left[\sin\!\left(\frac{\pi t}{2t_f}\right)\right]_0^{t_f}
  = g\cdot\frac{2t_f}{\pi}(\sin(\pi/2) - \sin(0)) = \frac{2g\,t_f}{\pi}
\]
For $t_f = 150\;\text{s}$: $\Delta v_\text{grav} = \frac{2\times 9.807 \times 150}{\pi} \approx 936\;\text{m/s}$.
\end{theorybox}

\begin{exercisebox}{1 — Tsiolkovsky: Step-by-Step Derivation Verification}
Redo the full derivation with $v_0 = 0$ and verify each integration step.

\textbf{a)} Separate variables from $m\,dv + v_e\,dm = 0$:
\[
  dv = -v_e\cdot\frac{dm}{m}
\]
Integrate both sides with correct limits ($v: 0 \to \Delta v$, $m: m_0 \to m_f$):
\[
  \int_0^{\Delta v} dv = -v_e\int_{m_0}^{m_f}\frac{dm}{m}
\]
Show every step of the right-hand integral.

\textbf{b)} The integral $\int_{m_0}^{m_f}\frac{dm}{m} = [\ln m]_{m_0}^{m_f} = \ln m_f - \ln m_0 = \ln\frac{m_f}{m_0}$\\
Therefore: $\Delta v = -v_e\ln\frac{m_f}{m_0} = v_e\ln\frac{m_0}{m_f}$.\\
Why does the sign flip? Write the argument carefully.
\end{exercisebox}

\begin{exercisebox}{3 — Gravity Loss: Analytical Integration}
For constant $g=9.807\;\text{m/s}^2$ and a linear pitch program from $\theta=90°$ to $\theta=0°$ over $t_f = 180\;\text{s}$:
\[
  \theta(t) = \frac{\pi}{2}\left(1 - \frac{t}{t_f}\right) \Rightarrow \sin\theta(t) = \sin\!\left(\frac{\pi}{2} - \frac{\pi t}{2t_f}\right) = \cos\!\left(\frac{\pi t}{2t_f}\right)
\]

\textbf{a)} Set up the integral:
\[
  \Delta v_\text{grav} = g\int_0^{t_f}\cos\!\left(\frac{\pi t}{2t_f}\right)dt
\]

\textbf{b)} Use substitution $u = \dfrac{\pi t}{2t_f}$, so $du = \dfrac{\pi}{2t_f}dt$, i.e.\ $dt = \dfrac{2t_f}{\pi}du$:
\[
  = g\cdot\frac{2t_f}{\pi}\int_0^{\pi/2}\cos(u)\,du
  = g\cdot\frac{2t_f}{\pi}\Big[\sin u\Big]_0^{\pi/2}
  = g\cdot\frac{2t_f}{\pi}\cdot\underline{\hspace{2cm}}
\]

\textbf{c)} Final result for $t_f = 180\;\text{s}$:
\[
  \Delta v_\text{grav} = \frac{2 \times 9.807 \times 180}{\pi} = \underline{\hspace{3cm}}\;\text{m/s}
\]

\textbf{d)} Verify numerically using the \textbf{trapezoid rule} with 3 points ($t=0, 90, 180$):
\[
  \Delta v_\text{grav} \approx \frac{180}{2}\cdot\frac{\cos(0) + 2\cos(\pi/4) + \cos(\pi/2)}{2} \cdot g = \underline{\hspace{3cm}}
\]
How large is the integration error of the trapezoid rule?
\end{exercisebox}

\begin{exercisebox}{5 — The Oberth Effect (Challenge)}
The Oberth effect: engine burns are most efficient at high speed. Formally:
the energy $\Delta E$ added by a burn producing $\Delta v$ at speed $v$ is:
\[
  \Delta E = \frac{1}{2}m(v + \Delta v)^2 - \frac{1}{2}mv^2 = m\,v\,\Delta v + \frac{1}{2}m(\Delta v)^2
\]

\textbf{a)} For the same $\Delta v = 100\;\text{m/s}$, $m=1000\;\text{kg}$:
Compute $\Delta E$ at $v = 0$ and at $v = 7000\;\text{m/s}$.\\
How many times more energy is added at $v=7000$ compared to $v=0$?

\textbf{b)} This is why the stage 2 burn (at high altitude, high speed) is so valuable.
For a rocket arriving at apoapsis with $v = 3000\;\text{m/s}$ and needing to reach $v_\text{circ} = 7700\;\text{m/s}$:
\[
  \Delta v = 7700 - 3000 = \underline{\hspace{2cm}}\;\text{m/s}
\]
What mass ratio does stage 2 need? ($v_e = 3500\;\text{m/s}$, $I_{sp,\text{vac}} = 357\;\text{s}$)

\textbf{c)} If instead the rocket \textit{first} circularises at a low orbit (small $\Delta v$) and \textit{then} raises apoapsis,
is that more or less efficient? Explain using the Oberth effect.
\end{exercisebox}
